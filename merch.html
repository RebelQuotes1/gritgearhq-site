from PIL import Image, ImageDraw, ImageFont # Create a canvas for the merch store placeholder width, height = 1920, 1080 canvas = Image.new("RGBA", (width, height), (15, 15, 15, 255)) # Dark grunge background draw = ImageDraw.Draw(canvas) # Load font font_path = "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf" title_font = ImageFont.truetype(font_path, 120) sub_font = ImageFont.truetype(font_path, 70) # Text content title_text = "MERCH COMING SOON" sub_text = "Because looking dangerous is half the battle." # Calculate text positions title_bbox = draw.textbbox((0, 0), title_text, font=title_font) title_x = (width - (title_bbox[2] - title_bbox[0])) // 2 title_y = 380 sub_bbox = draw.textbbox((0, 0), sub_text, font=sub_font) sub_x = (width - (sub_bbox[2] - sub_bbox[0])) // 2 sub_y = title_y + 160 # Draw text draw.text((title_x, title_y), title_text, font=title_font, fill=(255, 102, 0, 255)) # Orange draw.text((sub_x, sub_y), sub_text, font=sub_font, fill=(220, 220, 220, 255)) # Light gray # Save image canvas.save("Merch_ComingSoon_Placeholder.png", "PNG")
